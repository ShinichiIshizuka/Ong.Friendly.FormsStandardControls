using Codeer.Friendly.Dynamic;
using Codeer.Friendly.Windows;
using Codeer.Friendly.Windows.Grasp;
using Codeer.TestAssistant.GeneratorToolKit;
using Ong.Friendly.FormsStandardControls;

namespace TestCode
{
    [UserControlDriver(TypeFullName = "CreateDriverTarget.UserControl2")]
    public class UserControl2_Driver
    {
        public WindowControl Core { get; }
        public FormsButton button1 => new FormsButton(Core.Dynamic().button1);
        public FormsCheckBox checkBox1 => new FormsCheckBox(Core.Dynamic().checkBox1);
        public FormsTabControl tabControl1 => new FormsTabControl(Core.Dynamic().tabControl1);
        public UserControl1_Driver userControl11 => new UserControl1_Driver(new WindowControl(Core.Dynamic().userControl11));
        public FormsTextBox textBox1 => new FormsTextBox(Core.Dynamic().textBox1);

        public UserControl2_Driver(WindowControl core)
        {
            Core = core;
        }
    }

    public static class UserControl2_Driver_Extensions
    {
        [UserControlDriverIdentify]
        public static UserControl2_Driver Attach_UserControl2(this TabUserControlForm_Driver window)
            => new UserControl2_Driver(new WindowControl(window.Core.IdentifyFromTypeFullName("CreateDriverTarget.UserControl2")));
    }
}
